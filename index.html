// JavaScript to manage service listings with localStorage persistence const serviceForm = document.getElementById('service-form'); const serviceList = document.getElementById('service-list'); // Load services from localStorage function loadServices() { const services = JSON.parse(localStorage.getItem('services')) || []; services.forEach(service => addServiceToList(service)); } // Add service to the list function addServiceToList(service) { const li = document.createElement('li'); li.textContent = service; serviceList.appendChild(li); } // Handle form submission serviceForm.addEventListener('submit', (event) => { event.preventDefault(); const serviceName = event.target.elements['service-name'].value; addServiceToList(serviceName); // Save service to localStorage const services = JSON.parse(localStorage.getItem('services')) || []; services.push(serviceName); localStorage.setItem('services', JSON.stringify(services)); serviceForm.reset(); }); // Load services on page load loadServices();
